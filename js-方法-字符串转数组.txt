一、字符串 转 数组
1、split() 方法
	const text = "abc"
	const chars = text.split("")
	console.log(chars) // ['a', 'b', 'c']
2、展开运算符
	const text = "abc???"
	const chars = [...text]
	console.log(chars) //  ['a', 'b', 'c', '?', '?', '?']
3、解构赋值
	const text = "abc???"
	const [...chars] = text
	console.log(chars) //  ['a', 'b', 'c', '?', '?', '?']
4、Array.from() 
	const text = "abc"
	const chars = Array.from(text)
	console.log(chars) //  ['a', 'b', 'c']
